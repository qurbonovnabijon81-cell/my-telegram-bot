from telegram import Update, ReplyKeyboardMarkup
from telegram.ext import ApplicationBuilder, CommandHandler, MessageHandler, filters, ContextTypes

TOKEN = "8679650778:AAH7QNhpLAiemZFSfDgZny08ATe5ESEeXEQ"
ADMIN_ID = 7642515423  # <-- o'zing Telegram ID yoz

movies = {}

AD = "📢 Kanalga obuna bo‘ling: "@CineMax1bot"

keyboard = [
    ["🎬 Kino qidirish", "➕ Kino qo‘shish"],
    ["🔥 Kinolar", "ℹ️ Yordam"]
]
markup = ReplyKeyboardMarkup(keyboard, resize_keyboard=True)

user_state = {}

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("Salom! PRO Kino Bot 🎬\n\n" + AD, reply_markup=markup)

async def handle(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.message.from_user.id
    text = update.message.text

    # ➕ ADMIN kino qo‘shish rejimi
    if user_state.get(user_id) == "add_movie_name":
        user_state[user_id] = "add_movie_link_temp"
        user_state[user_id+"_name"] = text
        await update.message.reply_text("Endi kino linkini yuboring:")
        return

    if user_state.get(user_id) == "add_movie_link_temp":
        name = user_state.get(user_id+"_name")
        movies[name.lower()] = text
        user_state[user_id] = None
        await update.message.reply_text(f"✅ Kino qo‘shildi: {name}")
        return

    # 🔐 ADMIN PANEL
    if text == "➕ Kino qo‘shish":
        if user_id != ADMIN_ID:
            await update.message.reply_text("❌ Siz admin emassiz")
            return
        user_state[user_id] = "add_movie_name"
        await update.message.reply_text("Kino nomini yozing:")
        return

    # 🎬 Kino qidirish
    if text == "🎬 Kino qidirish":
        await update.message.reply_text("Kino nomini yozing:")
        return

    # 🔥 kinolar ro'yxati
    if text == "🔥 Kinolar":
        if movies:
            await update.message.reply_text("\n".join(movies.keys()))
        else:
            await update.message.reply_text("Hozircha kino yo‘q ❌")
        return

    # 🎥 kino topish
    key = text.lower()
    if key in movies:
        await update.message.reply_text(f"🎥 Kino: {movies[key]}")
    else:
        await update.message.reply_text("Kino topilmadi ❌")

app = ApplicationBuilder().token(TOKEN).build()

app.add_handler(CommandHandler("start", start))
app.add_handler(MessageHandler(filters.TEXT, handle))

app.run_polling()
